import spacy
import string
from pathlib import Path

# Load SpaCy model
try:
    nlp = spacy.load("en_core_web_sm")
except OSError:
    print("SpaCy English model not found.")
    print("Run:")
    print("python -m spacy download en_core_web_sm")
    exit()


def preprocess_text(text):
    """
    Cleans text by:
    - converting to lowercase
    - removing punctuation
    - removing stopwords
    """

    doc = nlp(text.lower())

    cleaned_words = [
        token.lemma_
        for token in doc
        if not token.is_stop
        and not token.is_punct
        and not token.is_space
    ]

    return " ".join(cleaned_words)


def calculate_similarity(text1, text2):

    processed_text1 = preprocess_text(text1)
    processed_text2 = preprocess_text(text2)

    doc1 = nlp(processed_text1)
    doc2 = nlp(processed_text2)

    return doc1.similarity(doc2)


def plagiarism_level(score):

    if score >= 0.85:
        return "HIGH"

    elif score >= 0.60:
        return "MEDIUM"

    else:
        return "LOW"


def generate_report(text1, text2):

    similarity = calculate_similarity(text1, text2)

    percentage = similarity * 100

    print("\n========== PLAGIARISM REPORT ==========")

    print(f"Similarity Score : {similarity:.2f}")
    print(f"Similarity       : {percentage:.2f}%")
    print(f"Plagiarism Level : {plagiarism_level(similarity)}")

    print("=======================================\n")


def main():

    print("====== Python Plagiarism Checker ======\n")

    text1 = input("Enter First Text:\n\n")

    print()

    text2 = input("Enter Second Text:\n\n")

    if not text1.strip() or not text2.strip():
        print("Error: Empty text is not allowed.")
        return

    generate_report(text1, text2)


if __name__ == "__main__":
    main()
